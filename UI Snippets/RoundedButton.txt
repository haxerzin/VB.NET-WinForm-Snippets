
Public Shared Sub CreateRoundedButton(ByRef btn As Button, ByVal radius As Integer)
    Dim path As New GraphicsPath()
    path.AddArc(New Rectangle(0, 0, radius, radius), 180, 90)
    path.AddArc(New Rectangle(btn.Width - radius, 0, radius, radius), -90, 90)
    path.AddArc(New Rectangle(btn.Width - radius, btn.Height - radius, radius, radius), 0, 90)
    path.AddArc(New Rectangle(0, btn.Height - radius, radius, radius), 90, 90)
    path.CloseFigure()
    btn.Region = New Region(path)
End Sub

// optimized 1 //

Public Shared Sub CreateRoundedButton(ByRef btn As Button, ByVal radius As Integer)
    btn.Region = New Region(New Rectangle(0, 0, btn.Width, btn.Height))
    btn.Region.Exclude(New Rectangle(radius, 0, btn.Width - (radius * 2), btn.Height))
    btn.Region.Exclude(New Rectangle(0, radius, btn.Width, btn.Height - (radius * 2)))
End Sub

// optimized 2 //

Public Shared Sub CreateRoundedButton(ByRef btn As Button, ByVal radius As Integer)
    btn.Region = New Region(New Rectangle(radius, radius, btn.Width - (radius * 2), btn.Height - (radius * 2)))
End Sub
